\section{Problema y objetivo}

\begin{frame}
  \frametitle{Problema}
  \begin{block}{Brecha principal}
    Convertir literatura científica y descripciones de paradigmas de decisión
    en modelos ejecutables sigue siendo una tarea manual, lenta y difícil de
    auditar.
  \end{block}

  \vspace{0.3cm}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Riesgos}
        \begin{itemize}
          \item Interpretaciones no trazables.
          \item Código separado de la formulación.
          \item Repetición de trabajo entre ejecuciones.
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Oportunidad}
        Usar agentes inteligentes con revisión humana, memoria y pruebas para
        convertir conocimiento científico en modelos simulables.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}
  \frametitle{Objetivos}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \begin{block}{Objetivo general}
        Diseñar e implementar un sistema de agentes que genere modelos de
        decisión ejecutables a partir de problemas descritos en lenguaje natural.
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{itemize}
        \item Investigar paradigmas científicos.
        \item Formalizarlos matemáticamente.
        \item Adaptarlos al entorno.
        \item Generar código Python probado.
        \item Reutilizar conocimiento aceptado.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Arquitectura}

\begin{frame}
  \frametitle{Idea central}
  \begin{center}
  \includegraphics[height=0.40\textheight]{figuras/phase1-main-transformation}
  \end{center}

  \vspace{0.4cm}
  \begin{block}{Tesis de diseño}
    No se usa un agente monolítico: se usa un pipeline trazable, con artefactos
    persistentes y revisión humana entre decisiones semánticas.
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Router como máquina de estados}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \begin{block}{Responsabilidad del Router}
        \begin{itemize}
          \item Coordinar etapas.
          \item Persistir estado y trazas.
          \item Pedir revisión humana.
          \item Activar memoria tras aceptación.
          \item Permitir reejecución parcial.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \scriptsize
      \begin{tabular}{ll}
        \textcolor{DecisionGreen}{$\bullet$} & CLASSIFY\_UMBRELLA\\
        \textcolor{DecisionGreen}{$\bullet$} & RESEARCH\\
        \textcolor{DecisionBlue}{$\bullet$} & REVIEW\_RESEARCH\\
        \textcolor{DecisionGreen}{$\bullet$} & MEMORY\_RESEARCH\\
        \textcolor{DecisionAmber}{$\bullet$} & FORMALIZE\\
        \textcolor{DecisionBlue}{$\bullet$} & REASON\\
        \textcolor{DecisionGreen}{$\bullet$} & BUILD\\
        \textcolor{DecisionGreen}{$\bullet$} & DONE
      \end{tabular}
      \vspace{0.3cm}

      \begin{exampleblock}{Clave}
        Los agentes generan contenido; el Router mantiene control operacional.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}
  \frametitle{Agentes especializados}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{Researcher}
        Identifica paradigmas, consulta memoria y lanza investigación profunda.
      \end{block}
      \begin{block}{Formalizer}
        Convierte informes científicos en formulaciones matemáticas comparables.
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{Reasoner}
        Adapta formulaciones al entorno y produce especificaciones JSON.
      \end{block}
      \begin{block}{Builder}
        Genera código Python autocontenido y tests ejecutables.
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.15cm}
  \begin{center}
    \scriptsize
    \texttt{deep/*.md} $\rightarrow$ \texttt{formulations/*.md} $\rightarrow$
    \texttt{reasoner/*.json} $\rightarrow$ \texttt{builder/*\_model.py}
  \end{center}
\end{frame}

\begin{frame}
  \frametitle{Infraestructura y artefactos}
  \begin{table}
  \scriptsize
  \begin{tabular}{lll}
    \toprule
    Capa & Tecnología & Responsabilidad\\
    \midrule
    Artefactos & MinIO & Informes, specs, código, estado y trazas\\
    Metadatos & PostgreSQL & Runs, modelos y ciclo de vida de memoria\\
    Grafo & Neo4j & Paradigmas, papers, variables y relaciones\\
    Recuperación & Qdrant & Hechos densos y BM25\\
    \bottomrule
  \end{tabular}
  \end{table}

  \begin{block}{Decisión importante}
    Los artefactos completos se guardan en MinIO. PostgreSQL conserva lo
    consultable; Neo4j representa relaciones; Qdrant recupera hechos.
  \end{block}
\end{frame}

    \begin{column}{0.52\textwidth}

        \begin{itemize}
          \item Densa: similitud semántica.
          \item BM25: términos exactos, autores y símbolos.
          \item Grafo: relaciones entre entidades.
          \item RRF: fusión sin entrenamiento.
          \item CRAG: corrección cuando hay duda.

      \end{block}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{exampleblock}{Flujo}
        \scriptsize
        Consulta\\
        $\downarrow$\\
        Qdrant denso + Qdrant BM25 + Neo4j\\
        $\downarrow$\\
        RRF + reranking\\
        $\downarrow$\\
        usar, complementar o fallback web
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Implementación y validación}

\begin{frame}[fragile]
  \frametitle{Contrato del modelo generado}
  \begin{block}{Salida de la fase 1}
    Modelos Python autocontenidos, cargables por duck typing desde el
    laboratorio de simulación.
  \end{block}

\begin{lstlisting}[language=Python]
def decide(self, perception: dict) -> Action: ...
def update(self, action, reward, new_perception) -> None: ...
def get_state(self) -> dict: ...
\end{lstlisting}

  \begin{alertblock}{Regla}
    \texttt{decide} elige acción sin mutar estado; \texttt{update} aprende
    después de recompensa y nueva percepción.
  \end{alertblock}
\end{frame}

\begin{frame}
  \frametitle{Validación}
  \begin{columns}
